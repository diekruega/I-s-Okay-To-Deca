            <!-- Prompts will appear here -->

        </div>

    <script>

        document.addEventListener('DOMContentLoaded', () => {

            const promptForm = document.getElementById('promptForm');

            const promptInput = document.getElementById('promptInput');

            const promptList = document.getElementById('promptList');


            promptForm.addEventListener('submit', (e) => {

                e.preventDefault();


                const promptText = promptInput.value.trim();

                if (!promptText) {

                    alert('Prompt cannot be empty!');

                    return;

                }


                const promptElement = document.createElement('div');

                promptElement.className = 'prompt';


                const promptContent = document.createElement('p');

                promptContent.textContent = promptText;


                const timestamp = document.createElement('div');

                timestamp.className = 'timestamp';

                const date = new Date();

                timestamp.textContent = `${date.toLocaleDateString()} ${date.toLocaleTimeString()}`;


                promptElement.appendChild(promptContent);

                promptElement.appendChild(timestamp);


                promptList.prepend(promptElement);

                promptInput.value = '';

            });

        });

    </script>
